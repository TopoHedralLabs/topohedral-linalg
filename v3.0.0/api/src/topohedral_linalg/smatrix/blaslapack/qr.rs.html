1//! QR decomposition of an [`SMatrix`] via LAPACK `dgeqrf`/`sgeqrf` and `dorgqr`/`sorgqr`.
2//!
3//! Provides the `qr()` method on [`SMatrix<T, N, M>`], computing A = QR with Q orthogonal and R
4//! upper-triangular. The static `Return<T, N, M>` struct carries Q and R as [`SMatrix`] values
5//! with appropriate compile-time dimensions. The implementation mirrors its [`DMatrix`](crate::dmatrix::DMatrix) counterpart
6//! but operates entirely on stack-allocated storage.
7//--------------------------------------------------------------------------------------------------
8
9//{{{ crate imports
10use crate::blaslapack::AsI32;
11use crate::blaslapack::{qr_raw, Geqrf, Orgqr, QrRawError};
12use crate::common::{Field, One, Zero};
13use crate::smatrix::SMatrix;
14//}}}
15//{{{ dep imports
16use thiserror::Error;
17//}}}
18//--------------------------------------------------------------------------------------------------
19
20//{{{ enum: Error
21/// Errors that can occur during QR decomposition.
22#[derive(Error, Debug)]
23pub enum Error
24{
25    /// Wraps a LAPACK `geqrf`/`orgqr` error from the QR factorisation steps.
26    #[error("Error in qr(), exited with error:\n{0}")]
27    GetrfError(#[from] QrRawError),
28}
29//}}}
30//{{{ struct: Return
31/// Result of a QR decomposition: orthogonal factor Q and upper-triangular factor R.
32pub struct Return<T, const N: usize, const M: usize>
33where
34    [(); N * M]:,
35    T: Field + Copy,
36{
37    /// Orthogonal factor Q.
38    pub q: SMatrix<T, N, M>,
39    /// Upper-triangular factor R.
40    pub r: SMatrix<T, N, M>,
41}
42//}}}
43//{{{ impl: SMatrix<T, N, M>
44#[allow(private_bounds)]
45impl<T, const N: usize, const M: usize> SMatrix<T, N, M>
46where
47    [(); N * M]:,
48    T: One + Zero + Geqrf + Orgqr + Field + Copy + AsI32,
49{
50    /// Computes the QR decomposition of the matrix, returning Q (orthogonal) and R (upper-triangular).
51    ///
52    /// # Errors
53    ///
54    /// Returns an error if either the LAPACK `geqrf` or `orgqr` routine fails.
55    pub fn qr(&self) -> Result<Return<T, N, M>, Error>
56    {
57        let raw = qr_raw(self.data.to_vec(), N, M)?;
58        let q_arr: [T; N * M] = raw.q_data.try_into().unwrap_or_else(|_| unreachable!());
59        let r_arr: [T; N * M] = raw.r_data.try_into().unwrap_or_else(|_| unreachable!());
60        Ok(Return {
61            q: SMatrix {
62                data: q_arr,
63                nrows: N,
64                ncols: M,
65            },
66            r: SMatrix {
67                data: r_arr,
68                nrows: N,
69                ncols: M,
70            },
71        })
72    }
73}
74//}}}
