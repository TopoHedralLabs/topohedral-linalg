1//! LAPACK `dgees`/`sgees` wrapper for the Schur decomposition.
2//!
3//! Provides the [`Gees`] trait, wrapping the LAPACK `?gees` routine that computes the Schur form
4//! of a real general matrix, optionally reordering the Schur form so that selected eigenvalues
5//! appear first. The `job` flag controls whether Schur vectors (Q) are computed; `sort` enables
6//! eigenvalue reordering. This is the LAPACK driver used by both `DMatrix::schur()` and
7//! `SMatrix::schur()`.
8//--------------------------------------------------------------------------------------------------

19//{{{ enum: Error
20/// Errors returned by the [`Gees`] LAPACK wrapper.
21#[derive(Error, Debug)]
22pub enum Error {
23    /// LAPACK returned a non-zero info code indicating the Schur decomposition failed.
24    #[error("Error in gees, exited with code {0}")]
25    LapackError(i32),
26}
27//}}}

29//{{{ trait: Gees
30/// Trait for types that support the Schur decomposition of a real general matrix.
31#[allow(clippy::too_many_arguments)]
32pub trait Gees: Copy {
33    /// Computes the Schur form of a real general matrix, optionally computing Schur vectors.
34    fn gees(
35        jobvs: u8,
36        sort: u8,
37        n: i32,
38        a: &mut [Self],
39        lda: i32,
40        sdim: &mut i32,
41        wr: &mut [Self],
42        wi: &mut [Self],
43        vs: &mut [Self],
44        ldvs: i32,
45        work: &mut [Self],
46        lwork: i32,
47        bwork: &mut [i32],
48    ) -> Result<(), Error>;
49}
50//}}}
51
52//{{{ impl: Gees for f64
53impl Gees for f64 {
54    #[inline]
55    fn gees(
56        jobvs: u8,
57        sort: u8,
58        n: i32,
59        a: &mut [Self],
60        lda: i32,
61        sdim: &mut i32,
62        wr: &mut [Self],
63        wi: &mut [Self],
64        vs: &mut [Self],
65        ldvs: i32,
66        work: &mut [Self],
67        lwork: i32,
68        bwork: &mut [i32],
69    ) -> Result<(), Error> {
70        let mut info = 0;
71        unsafe {
72            lapack::dgees(
73                jobvs, sort, None, n, a, lda, sdim, wr, wi, vs, ldvs, work, lwork, bwork, &mut info,
74            );
75        }
76        if info != 0 {
77            return Err(Error::LapackError(info));
78        }
79        Ok(())
80    }
81}
82//}}}

122//{{{ fun: schur_raw
123/// Shared GEES algorithm. Consumes the cloned matrix data; returns raw Q/T buffers.
124pub(crate) fn schur_raw<T>(
125    mut a_data: Vec<T>,
126    n: usize,
127    m: usize,
128) -> Result<SchurRaw<T>, Error>
129where
130    T: Gees + crate::common::One + crate::common::Zero + crate::common::Field + Default + Copy,
131{
132    let mut vs = vec![T::zero(); n * m];
133    let mut wr = vec![T::zero(); n];
134    let mut wi = vec![T::zero(); n];
135    let mut sdim = 0;
136    let mut work = vec![T::zero(); n * 5];
137    let lwork = (n * 5) as i32;
138    let mut bwork = vec![0; n];
139    T::gees(
140        b'V',
141        b'N',
142        n as i32,
143        &mut a_data,
144        n as i32,
145        &mut sdim,
146        &mut wr,
147        &mut wi,
148        &mut vs,
149        n as i32,
150        &mut work,
151        lwork,
152        &mut bwork,
153    )?;
154    Ok(SchurRaw {
155        q_data: vs,
156        t_data: a_data,
157    })
158}
