1//! Construction helpers for [`SMatrix`]: zeros, ones, fill, and slice initialisation.
2//!
3//! Provides factory methods on [`SMatrix<T, N, M>`] for the most common initialisation patterns.
4//! `zeros` and `ones` fill every element with the additive and multiplicative identities.
5//! `from_value` fills with a caller-supplied constant. `from_col_slice` accepts a flat slice in
6//! column-major order; `from_row_slice` accepts row-major input and transposes it to the internal
7//! column-major layout. When the `rand` feature is enabled, additional constructors accept
8//! distributions from the `rand` crate to produce randomly populated matrices.
9//!
10//--------------------------------------------------------------------------------------------------
11
12//{{{ crate imports
13use super::SMatrix;
14use crate::common::{One, Zero};
15//}}}
16//{{{ std imports
17//}}}
18//{{{ dep imports
19use rand::distr::{uniform::SampleUniform, Distribution, Uniform};
20//}}}
21//--------------------------------------------------------------------------------------------------
22
23//{{{ impl: SMatrix
24impl<T, const N: usize, const M: usize> SMatrix<T, N, M>
25where
26    T: Copy,
27{
28    //{{{ fun: zeros
29    /// Creates a new matrix with all elements set to zero.
30    pub fn zeros() -> Self
31    where
32        T: Zero,
33    {
34        Self {
35            data: [[T::zero(); N]; M],
36            nrows: N,
37            ncols: M,
38        }
39    }
40    //}}}
41    //{{{ fun: ones
42    /// Creates a new matrix with all elements set to one.
43    pub fn ones() -> Self
44    where
45        T: One,
46    {
47        Self {
48            data: [[T::one(); N]; M],
49            nrows: N,
50            ncols: M,
51        }
52    }
53    //}}}
54    //{{{ fun: from_value
55    /// Creates a new matrix with every element set to `value`.
56    pub fn from_value(value: T) -> Self {
57        Self {
58            data: [[value; N]; M],
59            nrows: N,
60            ncols: M,
61        }
62    }
63    //}}}
64    //{{{ fun: from_row_slice
65    /// Takes N*M element array in row-major order and creates a new SMatrix
66    pub fn from_row_slice(slice: &[T]) -> Self {
67        assert_eq!(slice.len(), N * M);
68
69        Self {
70            data: std::array::from_fn(|col| std::array::from_fn(|row| slice[row * M + col])),
71            nrows: N,
72            ncols: M,
73        }
74    }
75    //}}}
76    //{{{ fun: from_col_slice
77    /// Takes N*M element array in column-major order and creates a new SMatrix
78    pub fn from_col_slice(slice: &[T]) -> Self {
79        assert_eq!(slice.len(), N * M);
80        Self {
81            data: std::array::from_fn(|col| std::array::from_fn(|row| slice[col * N + row])),
82            nrows: N,
83            ncols: M,
84        }
85    }
86    //}}}
87    //{{{ fun: from_col_vec
88    pub(crate) fn from_col_vec(data: Vec<T>) -> Self {
89        assert_eq!(data.len(), N * M);
90        Self::from_col_slice(&data)
91    }
92    //}}}
93    //{{{ fun: from_uniform_random
94    /// Creates a new `SMatrix` with elements initialized to random values within the given range.
95    ///
96    /// The `low` and `high` parameters specify the inclusive range of the random values.
97    /// The matrix is initialized using a uniform random distribution.
98    pub fn from_uniform_random(
99        low: T,
100        high: T,
101    ) -> Self
102    where
103        T: SampleUniform + Zero,
104    {
105        let mut out = Self::zeros();
106
107        let range = Uniform::<T>::new(low, high).unwrap();
108
109        let mut rng = rand::rng();
110
111        for value in out.as_mut_slice() {
112            *value = range.sample(&mut rng);
113        }
114
115        out
116    }
117    //}}}
118    //{{{ fun: identity
119    /// Creates a new `SMatrix` initialized as the identity matrix.
120    ///
121    /// The identity matrix is a square matrix with 1s on the main diagonal and 0s elsewhere.
122    /// The dimensions of the identity matrix are determined by the generic parameters `N` and `M`.
123    pub fn identity() -> Self
124    where
125        T: One + Zero,
126    {
127        let mut out = Self::zeros();
128        let l = N.min(M);
129        for i in 0..l {
130            out[(i, i)] = T::one()
131        }
132        out
133    }
134    //}}}
135}
136//}}}
