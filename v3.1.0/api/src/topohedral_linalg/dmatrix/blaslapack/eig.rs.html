1//! General (non-symmetric) eigendecomposition of a [`DMatrix`] via LAPACK `dgeev`/`sgeev`.
2//!
3//! Provides the `eig()` method on [`DMatrix<T>`], computing all eigenvalues and both the left
4//! and right eigenvectors of a general square matrix. The computation is delegated to the
5//! `Geev` LAPACK driver. Eigenvalues are returned as complex numbers even when the input is
6//! real-valued; eigenvector matrices are stored column-major in the `Return<T>` struct. LAPACK
7//! errors propagate as a typed `Error`.
8//--------------------------------------------------------------------------------------------------
9
10//{{{ crate imports
11use crate::blaslapack::AsI32;
12use crate::blaslapack::{eig_raw, EigRawError, Geev};
13use crate::common::{Complex, Field, One, Zero};
14use crate::dmatrix::DMatrix;
15//}}}
16//{{{ dep imports
17use thiserror::Error;
18//}}}
19//--------------------------------------------------------------------------------------------------
20
21//{{{ enum: Error
22/// Errors that can occur during general eigendecomposition.
23#[derive(Error, Debug)]
24pub enum Error
25{
26    #[error("Error in eig(), exited with error:\n{0}")]
27    /// LAPACK `geev` failed to compute eigenvalues or eigenvectors.
28    GeevError(#[from] EigRawError),
29}
30//}}}
31
32//{{{ struct: Return
33/// Represents the eigenvalue decomposition of a square matrix of size `N`.
34///
35/// The eigenvalue decomposition of a matrix `A` is a factorization of the form `A = PDP^-1`,
36/// where `P` is the matrix of right eigenvectors, `D` is the diagonal matrix of eigenvalues,
37/// and `P^-1` is the matrix of left eigenvectors.
38///
39/// This struct contains the left and right eigenvectors, as well as the eigenvalues, of the
40/// decomposition.
41#[derive(Debug)]
42pub struct Return<T>
43where
44    T: Field + Default + Copy,
45{
46    /// Matrix whose columns are the left eigenvectors of A.
47    pub left_eigvecs: DMatrix<T>,
48    /// Matrix whose columns are the right eigenvectors of A.
49    pub right_eigvecs: DMatrix<T>,
50    /// Eigenvalues as complex numbers (real part from `wr`, imaginary part from `wi`).
51    pub eigvals: Vec<Complex<T>>,
52}
53//}}}
54
55//{{{ impl DMatrix<T>
56#[allow(private_bounds)]
57impl<T> DMatrix<T>
58where
59    T: One + Zero + Geev + Field + Default + Copy + AsI32,
60{
61    /// Computes the eigendecomposition of a general square matrix.
62    ///
63    /// Returns all eigenvalues and both left and right eigenvectors of `self`. Eigenvalues are
64    /// represented as complex numbers even when the input matrix is real-valued.
65    ///
66    /// # Errors
67    ///
68    /// Returns [`Error::GeevError`] if the LAPACK `geev` routine fails.
69    pub fn eig(&self) -> Result<Return<T>, Error>
70    {
71        let n = self.nrows;
72        let raw = eig_raw(self.data.clone(), n)?;
73        Ok(Return {
74            left_eigvecs: DMatrix {
75                data: raw.vl,
76                nrows: n,
77                ncols: n,
78            },
79            right_eigvecs: DMatrix {
80                data: raw.vr,
81                nrows: n,
82                ncols: n,
83            },
84            eigvals: raw.eigvals,
85        })
86    }
87}
88//}}}
