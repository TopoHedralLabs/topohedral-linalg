1//! In-place element-wise transformation for [`SMatrix`].
2//!
3//! Implements the [`TransformOps`] trait for [`SMatrix<T, N, M>`], providing a `transform` method
4//! that applies a caller-supplied closure to every element in place in column-major order. No extra
5//! allocation is required because the transformation is purely in-place, making this an efficient
6//! complement to the lazy arithmetic operators for cases where the transformation cannot be
7//! expressed as a simple arithmetic expression.
8//--------------------------------------------------------------------------------------------------
9
10//{{{ crate imports
11use super::SMatrix;
12use crate::common::TransformOps;
13use crate::subviews::MatrixViewMut;
14//}}}
15//{{{ std imports
16//}}}
17//{{{ dep imports
18//}}}
19//--------------------------------------------------------------------------------------------------
20
21//{{{ impl: TransformOps for SMatrix
22impl<T, const N: usize, const M: usize> TransformOps for SMatrix<T, N, M>
23where
24    T: Copy,
25{
26    type ScalarType = T;
27
28    fn transform<F>(
29        &mut self,
30        mut f: F,
31    ) where
32        F: FnMut(Self::ScalarType) -> Self::ScalarType,
33    {
34        for value in self.as_mut_slice() {
35            *value = f(*value);
36        }
37    }
38}
39//}}}
40//{{{ impl: TransformOps for MatrixViewMut
41impl<'a, T, const N: usize, const M: usize> TransformOps for MatrixViewMut<'a, SMatrix<T, N, M>>
42where
43    T: Copy,
44{
45    type ScalarType = T;
46
47    fn transform<F>(
48        &mut self,
49        mut f: F,
50    ) where
51        F: FnMut(Self::ScalarType) -> Self::ScalarType,
52    {
53        for col in 0..self.ncols {
54            for row in 0..self.nrows {
55                let value = self[(row, col)];
56                (*self)[(row, col)] = f(value);
57            }
58        }
59    }
60}
61//}}}
