1//! Cholesky decomposition of a [`DMatrix`] via LAPACK `dpotrf`/`spotrf`.
2//!
3//! Provides the `cholesky()` method on square symmetric positive-definite [`DMatrix<T>`]
4//! instances, computing the lower-triangular factor `L` such that `A = L L^T`. The factorisation
5//! is delegated to LAPACK `Potrf`; failures, including non-positive-definite input, are reported
6//! through a typed `Error`.
7//--------------------------------------------------------------------------------------------------
8
9//{{{ crate imports
10use crate::blaslapack::{cholesky_raw, CholeskyRawError, Potrf};
11use crate::common::{Field, Zero};
12use crate::dmatrix::DMatrix;
13//}}}
14//{{{ dep imports
15use thiserror::Error;
16//}}}
17//--------------------------------------------------------------------------------------------------
18
19//{{{ enum: Error
20/// Errors that can occur during Cholesky decomposition.
21#[derive(Error, Debug)]
22pub enum Error {
23    #[error("Error in cholesky(), exited with error:\n{0}")]
24    /// LAPACK `potrf` reported a failure, e.g. the matrix is not positive definite.
25    PotrfError(#[from] CholeskyRawError),
26}
27//}}}
28//{{{ struct: Return
29/// Represents the Cholesky decomposition of a symmetric positive-definite matrix.
30#[derive(Debug)]
31pub struct Return<T>
32where
33    T: Field + Copy,
34{
35    /// Lower-triangular factor L such that A = L L^T.
36    pub l: DMatrix<T>,
37}
38//}}}
39//{{{ impl DMatrix<T>
40#[allow(private_bounds)]
41impl<T> DMatrix<T>
42where
43    T: Zero + Potrf + Field + Copy,
44{
45    /// Computes the Cholesky decomposition of the matrix.
46    ///
47    /// Factors `self` into `L` such that `self = L L^T`. Only the lower triangle of `self` is
48    /// read by LAPACK.
49    ///
50    /// # Errors
51    ///
52    /// Returns [`Error::PotrfError`] if LAPACK `potrf` fails, including when the matrix is not
53    /// positive definite.
54    pub fn cholesky(&self) -> Result<Return<T>, Error> {
55        let n = self.nrows;
56        if n != self.ncols {
57            panic!("Matrix must be square for Cholesky decomposition");
58        }
59        let raw = cholesky_raw(self.data.clone(), n)?;
60        Ok(Return {
61            l: DMatrix {
62                data: raw.l_data,
63                nrows: n,
64                ncols: n,
65            },
66        })
67    }
68}
69//}}}
